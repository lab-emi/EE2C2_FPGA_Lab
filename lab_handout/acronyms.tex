% use for normal acronyms, gives them a PDF tooltip popup on subsequent uses, e.g. \tip{aae}
\newcommand*{\tip}[1]{%  define our acronym command,  make it short since we use it a lot, use * for so that it is only a 'short' command
    \ifglsused{#1}{% if we used it already, then put pdftooltip
    %   {\tooltip{\accolor{\glsentryshort{#1}}}{\glsentrydesc{#1}}}%
      {\pdftooltip{\accolor{\glsentryshort{#1}}}{\glsentrydesc{#1}}}%
    }{%
      \gls{#1}% otherwise put the normal gls
    }%
}%

\newcommand*{\tipshort}[1]{%  define our acronym command,  make it short since we use it a lot, use * for so that it is only a 'short' command
      {\pdftooltip{\accolor{\glsentryshort{#1}}}{\glsentrydesc{#1}}}%
}%

% use for plural acronyms, e.g. \tips{aae}
\newcommand*{\tips}[1]{%  define our acronym command,  make it short since we use it a lot, use * for so that it is only a 'short' command
    \ifglsused{#1}{% if we used it already, then put pdftooltip
      {\pdftooltip{\accolor{\glsentryshortpl{#1}}}{\glsentrydescplural{#1}}}%
    %   {\tooltip{\accolor{\glsentryshortpl{#1}}}{\glsentrydesc{#1}}}%
    }{%
      \glspl{#1}% otherwise put the normal gls
    }%
}%
