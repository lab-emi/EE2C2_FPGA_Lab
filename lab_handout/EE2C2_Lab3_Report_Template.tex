\noindent\textbf{Checkpoint 1a: reading the logic.}
\begin{enumerate}[noitemsep]
  \item Identify the line that implements an \textbf{AND}, an \textbf{OR}, a \textbf{NOT}, an \textbf{XOR},
        and a \textbf{multiplexer}.
  \item \texttt{led\_comb[1] = sw[0] \& sw[1]} and \texttt{led\_comb[7] = \&sw} both use \texttt{\&}, yet

\section*{Topic 4 --- Static timing analysis}

\noindent\textbf{Checkpoint 4.}
\begin{enumerate}[noitemsep]
  \item Record \textbf{WNS} and \textbf{TNS}. Is setup timing met?
  \item Record the start point and endpoint of the critical path, and describe in one or two sentences what
        hardware lies between them.
  \item Could a \emph{hold} violation (negative WHS) be fixed by slowing the clock?
  \item (Optional) Open the timing report for the worst setup path and read off the clock period
        $T_{\mathrm{clk}}$ (the requirement), the \textbf{data path delay} ($t_{\mathrm{cq}} +
        t_{\mathrm{comb}}$), the \textbf{clock uncertainty} $t_{\mathrm{unc}}$, and the \textbf{slack}.
        Rearrange the setup-slack equation to solve for the setup-time requirement
        \[ t_{\mathrm{su}} = T_{\mathrm{clk}} - t_{\mathrm{unc}} - (t_{\mathrm{cq}} + t_{\mathrm{comb}})
           - t_{\mathrm{slack}}, \]
        and report your four numbers and the resulting $t_{\mathrm{su}}$.
\end{enumerate}
\answerbox{Your answer}{0.22\textheight}
